# 街舞分类模型（DanceStyle）
rails generate model DanceStyle name:string description:text image:string created_at:datetime updated_at:datetime


# app/models/dance_style.rb
class DanceStyle < ApplicationRecord
  # 图片上传配置
  mount_uploader :image, ImageUploader
  
  # 数据验证
  validates :name, presence: true, length: { maximum: 100 }
  validates :description, presence: true
  
  # 时间自动维护
  before_save :set_timestamps
  
  private
  
  def set_timestamps
    self.updated_at = Time.current
    self.created_at = Time.current if new_record?
  end
end

rails generate model Event title:string date:date description:text image:string created_at:datetime updated_at:datetime

# app/models/event.rb
class Event < ApplicationRecord
  mount_uploader :image, ImageUploader
  
  validates :title, presence: true, length: { maximum: 200 }
  validates :date, presence: true
  validates :description, presence: true
  
  # 按日期倒序排列
  default_scope { order(date: :desc) }
  
  before_save :set_timestamps
  
  private
  
  def set_timestamps
    self.updated_at = Time.current
    self.created_at = Time.current if new_record?
  end
end

# 关于我们模型（AboutUs）
rails generate model AboutUs mission:text teaching_philosophy:text image:string updated_at:datetime

# app/models/about_us.rb
class AboutUs < ApplicationRecord
  mount_uploader :image, ImageUploader
  
  validates :mission, presence: true
  validates :teaching_philosophy, presence: true
  
  # 确保只有一条记录（关于我们通常只有一份）
  def self.instance
    first || create!(mission: "", teaching_philosophy: "")
  end
  
  before_save :set_updated_at
  
  private
  
  def set_updated_at
    self.updated_at = Time.current
  end
end

rails generate model ContactInfo address:string phone:string email:string business_hours:string updated_at:datetime

# app/models/contact_info.rb
class ContactInfo < ApplicationRecord
  validates :address, presence: true
  validates :phone, presence: true
  validates :email, presence: true, format: { with: URI::MailTo::EMAIL_REGEXP }
  validates :business_hours, presence: true
  
  # 确保只有一条记录
  def self.instance

  end
  
  before_save :set_updated_at
  
  private
  
  def set_updated_at
    self.updated_at = Time.current
  end
end

# 图片上传配置（CarrierWave）
# 生成上传器
rails generate uploader Image

# app/uploaders/image_uploader.rb
# encoding: utf-8
class ImageUploader < CarrierWave::Uploader::Base
  # 使用MiniMagick处理图片
  include CarrierWave::MiniMagick

  # 存储方式：开发环境本地存储，生产环境可扩展到S3
  storage :file

  # 图片存储路径
  def store_dir
    "uploads/#{model.class.to_s.underscore}/#{mounted_as}/#{model.id}"
  end

  # 允许的图片格式
  def extension_allowlist
    %w(jpg jpeg gif png webp)
  end

  # 图片尺寸限制（可选，优化前端展示）
  version :thumb do
    process resize_to_fill: [200, 200]
  end

  version :medium do
    process resize_to_fit: [800, 600]
  end

  # 文件名处理（避免中文/特殊字符）
  def filename
    "#{secure_token}.#{file.extension}" if original_filename.present?
  end

  protected
  def secure_token
    var = :"@#{mounted_as}_secure_token"
    model.instance_variable_get(var) or model.instance_variable_set(var, SecureRandom.uuid)
  end
end

# Active Admin 后台配置
# app/admin/dance_styles.rb
ActiveAdmin.register DanceStyle do
  # 中文名称
  menu label: "街舞分类"
  
  # 后台显示的字段
  index do
    selectable_column
    column "舞种名称", :name
    column "描述", :description do |dance_style|
      truncate(dance_style.description, length: 50)
    end
    column "创建时间", :created_at
    column "更新时间", :updated_at
    actions
  end

  # 表单编辑字段
  form do |f|
    f.inputs "街舞分类信息" do
      f.input :name, label: "舞种名称"
      f.input :description, label: "舞种描述", input_html: { rows: 5 }
      f.input :image, label: "舞种图片", as: :file, hint: f.object.image.present? ? image_tag(f.object.image.medium.url) : ""
    end
    f.actions
  end

  # 详情页显示
  show do
    attributes_table do
      row :name
      row :description
      row :image do |dance_style|
        dance_style.image.present? ? image_tag(dance_style.image.medium.url) : "无图片"
      end
      row :created_at
      row :updated_at
    end
  end

  # 搜索功能
  filter :name, label: "舞种名称"
  filter :description, label: "描述"
end
